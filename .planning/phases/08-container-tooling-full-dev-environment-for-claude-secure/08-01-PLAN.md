Create `docker-compose.dev.yml` in the project root with the following exact content:

```yaml
# Development overrides -- use with: docker compose -f docker-compose.yml -f docker-compose.dev.yml up --watch
# Hot-reloads proxy.js and validator.py on file changes without image rebuilds.
#
# NOTE: validator uses network_mode: service:claude. If claude restarts,
# validator must also restart (handled automatically by sync+restart).
services:
  proxy:
    develop:
      watch:
        - action: sync+restart
          path: ./proxy/proxy.js
          target: /app/proxy.js

  validator:
    develop:
      watch:
        - action: sync+restart
          path: ./validator/validator.py
          target: /app/validator.py
```

This file ONLY adds `develop.watch` keys. It does NOT replace any existing environment, volumes, or network config from docker-compose.yml. The merge must be additive only.

Do NOT add watch config for the claude container -- it runs Claude Code CLI, there is no source code to hot-reload.

Create `Makefile` in the project root with the following targets. Use TAB indentation (mandatory for Makefiles). Each target has a `## description` comment for the help target.

```makefile
.PHONY: build up down dev test lint clean help

.DEFAULT_GOAL := help

build:  ## Build all Docker images
	docker compose build

up:  ## Start all services in background
	docker compose up -d

down:  ## Stop and remove all services
	docker compose down

dev:  ## Start services with hot-reload (watch mode)
	LOG_HOOK=1 LOG_ANTHROPIC=1 LOG_IPTABLES=1 \
	docker compose -f docker-compose.yml -f docker-compose.dev.yml up --watch

test:  ## Run all integration tests
	bash tests/run-all.sh

lint:  ## Lint compose files and bash scripts (via ShellCheck container)
	@echo "Validating docker-compose files..."
	docker compose config --quiet
	docker compose -f docker-compose.yml -f docker-compose.dev.yml config --quiet
	@echo "Running ShellCheck on bash scripts..."
	docker run --rm -v "$$(pwd):/mnt:ro" koalaman/shellcheck:stable \
	    /mnt/bin/claude-secure \
	    /mnt/claude/hooks/pre-tool-use.sh \
	    /mnt/install.sh \
	    /mnt/tests/test-phase1.sh \
	    /mnt/tests/test-phase2.sh \
	    /mnt/tests/test-phase3.sh \
	    /mnt/tests/test-phase4.sh \
	    /mnt/tests/test-phase6.sh \
	    /mnt/tests/test-phase7.sh

clean:  ## Remove containers, images, and volumes
	docker compose down -v --rmi local

help:  ## Show available targets
	@grep -E '^[a-zA-Z_-]+:.*?## .*$$' $(MAKEFILE_LIST) | \
	    awk 'BEGIN {FS = ":.*?## "}; {printf "  \033[36m%-15s\033[0m %s\n", $$1, $$2}'
```

Key details:
- `dev` target enables all logging env vars (LOG_HOOK=1 LOG_ANTHROPIC=1 LOG_IPTABLES=1) as recommended in research
- `lint` target validates BOTH compose files and runs containerized ShellCheck (no host dependency)
- `lint` target lists each bash script explicitly (not a glob) to avoid issues with ShellCheck container path resolution
- `test` target delegates to `tests/run-all.sh` (created in Plan 02)
- `.DEFAULT_GOAL := help` so running `make` with no arguments shows help
- ShellCheck uses `:ro` mount for security -- it only needs to read scripts
